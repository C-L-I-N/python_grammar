## 关键字参数传递
```python
def print_fullname(firstname, lastname):
    full_name = firstname + ' ' + lastname
    print(full_name)

# 顺序无关！
print_fullname(firstname='Asabeneh', lastname='Yetayeh')
print_fullname(lastname='Yetayeh', firstname='Asabeneh')  # 同样有效
```

## 默认参数
```python
def greetings(name='Peter'):
    message = name + ', welcome to Python for Everyone!'
    return message

print(greetings())           # Peter, welcome to Python for Everyone!
print(greetings('Asabeneh')) # Asabeneh, welcome to Python for Everyone!

def weight_of_object(mass, gravity=9.81):
    weight = str(mass * gravity) + ' N'
    return weight

print(weight_of_object(100))        # 981.0 N (地球)
print(weight_of_object(100, 1.62))  # 162.0 N (月球)
```


## 任意参数
```python
def sum_all_nums(*nums):
    total = 0
    for num in nums:
        total += num
    return total

print(sum_all_nums(2, 3, 5))        # 10
print(sum_all_nums(1, 2, 3, 4, 5))  # 15
```
